fun test {
  bb 0 {
    v0 = LoadConst<NoneType>
    CondBranch<1, 3> v0
  }
  bb 1 {
    v1 = Phi<0, 2> v0 v2
    Branch<2>
  }
  bb 2 {
    v2 = LoadConst<NoneType>
    CondBranch<1, 3> v1
  }
  bb 3 {
    v3 = Phi<0, 2> v0 v1
    Return v3
  }
}

    CondBranch<1, 3> v1

    v3:NoneType = Phi<0, 2> v0 v1
    Return v3
  }
}
